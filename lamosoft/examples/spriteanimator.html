spriteanimator.js 

let editSprite = new Sprite({
    el: 'canvas1',
    cols: 16,
    rows: 16
});
let editor = new SpriteEditor({
    sprite: editSprite
});
let animatedSprites = [];
for( var i = 1; i < 10; i++ ) {
    let animatedSprite = new Sprite({
        el: 'anim' + i,
        cols: 16,
        rows: 16
    });
    animatedSprites.push( animatedSprite );
}
let spriteAnimator = new SpriteAnimator({
    editor: editor,
    el: 'animatordiv',
    animatedSprites: animatedSprites,
    count: 16,
    persistenceId: 'examples.spriteAnimator'
});
